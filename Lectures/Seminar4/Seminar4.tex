\frame{
  \frametitle{Automatically Include Tables with xtable}
  {\Large{We can use the {\emph{xtable}} package and the \texttt{print} command to {\bf{automatically include}} tables in our R Markdown documents. \\[0.5cm]
  We can create tables from data frames and many other object types.
  }}
}

\begin{frame}[fragile]

\begin{knitrout}
\definecolor{shadecolor}{rgb}{0.969, 0.969, 0.969}\color{fgcolor}\begin{kframe}
\begin{alltt}
\hlcomment{# Load package}
\hlfunctioncall{library}(xtable)

\hlcomment{# Create data frame object}
Population <- \hlfunctioncall{c}(14.3, 6.3, 66.7)

Countries <- \hlfunctioncall{c}(\hlstring{"Cambodia"}, \hlstring{"Laos"}, \hlstring{"Thailand"})

NewData <- \hlfunctioncall{data.frame}(Countries, Population)
\end{alltt}
\end{kframe}
\end{knitrout}

\end{frame}
